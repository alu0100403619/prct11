def convert(array)
        @_fil = array.length
        @_col = array[0].length
        
        matriz_retorno = nil
        totalElementos = @_fil * @_col
        if (ContarCeros(array) % totalElementos) < 0.6
                matriz_retorno = Matriz_Densa.new(@_fil, @_col)
                matriz_retorno.copy!(array)
        else
                matriz_retorno = Matriz_Dispersa.new(@_fil, @_col)
                matriz_retorno.copy!(array)
        end   
        
        matriz_retorno
end
